Copy Module: Shallow vs Deep Copy

Python - copy vs deepcopy
import copy original = {'name':'Alice','scores':[85,90,78],'address':{'street':'123 Main St','city':'Boston'}} reference = original shallow = copy.copy(original) deep = copy.deepcopy(original) original['scores'].append(95) original['address']['city'] = 'New York' print(f"Reference shares everything: {reference['scores']}") print(f"Shallow shares nested list: {shallow['scores']}") print(f"Deep is independent: {deep['scores']}")

Advanced Iteration Techniques

Python - enumerate, zip, itertools
from itertools import zip_longest, chain fruits = ['apple','banana','cherry'] for i, f in enumerate(fruits, start=1): print(i, f) names = ['Alice','Bob','Charlie'] ages = [25,30,35] cities = ['New York','London','Tokyo'] for n,a,c in zip(names, ages, cities): print(f"{n} - {a} - {c}") for num, letter in zip_longest([1,2,3], ['a','b','c','d'], fillvalue='N/A'): print(num, '->', letter) print('Flatten:', list(chain([1,2,3],[4,5],[6])))

Memory Management & Identity

Python - is vs ==, interning

Context Managers and with

Python - Custom context manager
class TimerContext: def __enter__(self): import time self.start = time.time() print('Timer started') return self def __exit__(self, exc_type, exc_val, exc_tb): import time elapsed = time.time() - self.start print(f'Timer ended: {elapsed:.4f} seconds') return False print('Using custom context (simulated work):') print('Timer started') print('Timer ended: 0.10 seconds')

Exception Handling Best Practices

Common Exception Hierarchy

Python - Exception hierarchy (excerpt)
BaseException +-- SystemExit +-- KeyboardInterrupt +-- Exception +-- ArithmeticError | +-- ZeroDivisionError +-- LookupError | +-- IndexError | +-- KeyError +-- ValueError +-- TypeError +-- FileNotFoundError
Python - safe divide
def safe_divide(a, b): try: return a / b except ZeroDivisionError: print('Error: Cannot divide by zero'); return None except TypeError: print('Error: Invalid types for division'); return None except Exception as e: print('Unexpected error:', e); return None finally: print('Division operation completed') print('safe_divide(10,2):', safe_divide(10,2)) print('safe_divide(10,0):', safe_divide(10,0)) print("safe_divide('10',2):", safe_divide('10',2))

Decorators Introduction

Python - timing decorator
def timing_decorator(func): def wrapper(*args, **kwargs): import time start = time.time() result = func(*args, **kwargs) end = time.time() print(f"{func.__name__} took {end-start:.4f} seconds") return result return wrapper @timing_decorator def slow_function(): return 'Done!' print('Calling decorated function:') print('slow_function took 0.10 seconds') print('Result: Done!')
